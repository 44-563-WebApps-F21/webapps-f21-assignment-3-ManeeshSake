Breadth-first search algorithm

        vector> adj;  // adjacency list representation
            int n; // number of nodes
            int s; // source vertex
            
            queue q;
            vector used(n);
            vector d(n), p(n);
            
            q.push(s);
            used[s] = true;
            p[s] = -1;
            while (!q.empty()) {
                int v = q.front();
                q.pop();
                for (int u : adj[v]) {
                    if (!used[u]) {
                        used[u] = true;
                        q.push(u);
                        d[u] = d[v] + 1;
                        p[u] = v;
                    }
                }
            }
            
            if (!used[u]) {
                cout << "No path!";
            } else {
                vector path;
                for (int v = u; v != -1; v = p[v])
                    path.push_back(v);
                reverse(path.begin(), path.end());
                cout << "Path: ";
                for (int v : path)
                    cout << v << " ";
            }
